{{- /* One print: a polaroid figure. Takes an image dict {img, alt, caption}; optional keys: eager (skip lazy-loading, for above-the-fold images), view (opens in the plate viewer), full (larger file for the viewer), youtube (video id; img becomes the poster), video (a silent looping film; img becomes its poster), depth (position in a pile), sizes (how wide this print renders, for srcset selection). Content refers to originals under assets/img; the WebP variants the page actually serves are generated here, at build time. */ -}} {{- $orig := resources.Get (strings.TrimPrefix "/" .img) -}} {{- $srcset := slice -}} {{- range $w := slice 400 800 -}} {{- if lt $w $orig.Width -}} {{- $v := $orig.Process (printf "resize %dx webp q80" $w) -}} {{- $srcset = $srcset | append (printf "%s %dw" $v.RelPermalink $w) -}} {{- end -}} {{- end -}} {{- $best := $orig.Process "webp q80" -}} {{- $srcset = $srcset | append (printf "%s %dw" $best.RelPermalink $orig.Width) -}} {{- $sizes := .sizes | default "(max-width: 640px) 92vw, 330px" -}}
